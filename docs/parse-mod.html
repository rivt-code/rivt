149    def str_parse(self, strL):
150        """parse method string line by line starting with second line
151
152            :param list strL: split method string
153            :return mdS: md formatted string
154            :return rstS: reST formatted string
155            :return labelD: increment references
156            :return folderD: folder paths
157            :rtype mdS: string
158            :rtype rstS: string
159            :rtype folderD: dictionary
160            :rtype labelD: dictionary
161        """
162
163        xutfS = """"""      # utfS local string
164        xmdS = """"""       # mdS local string
165        xrstS = """"""      # rstS local string
166        uS = """"""         # raw local line
167        blockB = False
168
169        # table alignment
170        hdrdL = ["variable", "value", "[value]", "description"]
171        aligndL = ["left", "right", "right", "left"]
172        hdraL = ["variable", "value", "[value]", "description [eq. number]"]
173        alignaL = ["left", "right", "right", "left"]
174
175        blockevalL = []     # current value table
176        blockevalB = False  # stop accumulation of values
177        vtableL = []        # value table for export
178        eqL = []            # equation result table
179        lineS = ""
180        for uS in strL:
181            # print(f"{blockassignB=}")
182            # print(f"{uS=}")
183            uS = uS[4:]                                # remove indent
184            if blockB:                                 # accumulate block
185                lineS += uS
186                continue
187            if blockB and uS.strip() == "[q]]":        # end of block
188                tagS = self.tagsD["[q]"]
189                rvtS = tag_utf.TagsUTF(lineS, tagS,
190                                       self.labelD, self.folderD,  self.rivtD)
191                xutfS += rvtS + "\n"
192                rvtS = tag_md.TagsMD(lineS, tagS,
193                                     self.labelD, self.folderD,  self.rivtD)
194                xmdS += rvtS + "\n"
195                rvtS = tag_rst.TagsRST(lineS, tagS,
196                                       self.labelD, self.folderD,  self.rivtD)
197                xrstS += rvtS + "\n"
198                blockB = False
199            if blockevalB and len(uS.strip()) < 2:    # value tables
200                vtableL += blockevalL
201                if tfS == "declare":
202                    vutfS = self.dtable(blockevalL, hdrdL,
203                                        "rst", aligndL) + "\n\n"
204                    vmdS = self.dtable(blockevalL, hdrdL,
205                                       "html", aligndL) + "\n\n"
206                    xutfS += vutfS
207                    xmdS += vmdS
208                    xrstS += vutfS
209                if tfS == "assign":
210                    vutfS = self.dtable(blockevalL, hdrdL,
211                                        "rst", aligndL) + "\n\n"
212                    vmdS = self.atable(blockevalL, hdraL,
213                                       "html", alignaL) + "\n\n"
214                    xutfS += vutfS
215                    xmdS += vmdS
216                    xrstS += vutfS
217                blockevalL = []
218                blockevalB = False
219            elif uS[0:2] == "||":                      # commands
220                usL = uS[2:].split("|")
221                parL = usL[1:]
222                cmdS = usL[0].strip()
223                if cmdS in self.cmdL:
224                    rvtC = cmd_utf.CmdUTF(
225                        parL, self.labelD, self.folderD, self.rivtD)
226                    utfS = rvtC.cmd_parse(cmdS)
227                    # print(f"{utfS=}")
228                    xutfS += utfS
229                    rvtC = cmd_md.CmdMD(
230                        parL, self.labelD, self.folderD, self.rivtD)
231                    mdS = rvtC.cmd_parse(cmdS)
232                    # print(f"{mdS=}")
233                    xmdS += mdS
234                    rvtC = cmd_rst.CmdRST(
235                        parL, self.labelD, self.folderD, self.rivtD)
236                    reS = rvtC.cmd_parse(cmdS)
237                    xrstS += reS
238            elif "_[" in uS:                           # line tag
239                usL = uS.split("_[")
240                lineS = usL[0]
241                tagS = usL[1].strip()
242                if tagS[0] == "[":                     # block tag
243                    blockB = True
244                if tagS in self.tagsD:
245                    rvtC = tag_utf.TagsUTF(lineS, self.labelD, self.folderD,
246                                           self.tagsD, self.rivtD)
247                    utfxS = rvtC.tag_parse(tagS)
248                    xutfS += utfxS + "\n"
249                    rvtC = tag_md.TagsMD(lineS, self.labelD, self.folderD,
250                                         self.tagsD, self.rivtD)
251                    mdS = rvtC.tag_parse(tagS)
252                    xmdS += mdS + "\n"
253                    rvtC = tag_rst.TagsRST(lineS, self.labelD, self.folderD,
254                                           self.tagsD, self.rivtD)
255                    reS = rvtC.tag_parse(tagS)
256                    xrstS += reS + "\n"
257            elif "=" in uS and self.methS == "V":       # equation tag
258                # print(f"{uS=}")
259                usL = uS.split("|")
260                lineS = usL[0]
261                self.labelD["unitS"] = usL[1].strip()
262                self.labelD["descS"] = usL[2].strip()
263                rvtC = tag_md.TagsMD(lineS, self.labelD, self.folderD,
264                                     self.localD)
265                if ":=" in uS:                         # declare tag
266                    tfS = "declare"
267                    blockevalL.append(rvtC.tag_parse(":="))
268
269                    rvtC = tag_rst.TagsRST(lineS, self.labelD, self.folderD,
270                                           self.localD)
271                    eqL = rvtC.tag_parse(":=")
272                    blockevalB = True
273                    continue
274                else:
275                    tfS = "assign"                     # assign tag
276                    eqL = rvtC.tag_parse("=")
277                    mdS += eqL[1]
278                    blockevalL.append(eqL[0])
279
280                    rvtC = tag_rst.TagsRST(lineS, self.labelD, self.folderD,
281                                           self.localD)
282                    eqL = rvtC.tag_parse("=")
283                    rstS += eqL[1]
284                    blockevalB = True
285                    continue
286            else:
287                print(uS)       # pass unformatted string
288
289        # export values
290        valP = Path(self.folderD["dataP"], self.folderD["valfileS"])
291        with open(valP, "w", newline="") as f:
292            writecsv = csv.writer(f)
293            writecsv.writerow(hdraL)
294            writecsv.writerows(vtableL)

296        return (xutfS, xmdS, xrstS,  self.labelD, self.folderD, self.rivtD)

298    def atable(self, tblL, hdreL, tblfmt, alignaL):
299        """write assign values table"""
300
301        locals().update(self.rivtD)
302
303        valL = []
304        for vaL in tblL:
305
306            varS = vaL[0].strip()
307            valS = vaL[1].strip()
308            unit1S, unit2S = vaL[2], vaL[3]
309            descripS = vaL[4].strip()
310            if unit1S != "-":
311                if type(eval(valS)) == list:
312                    val1U = array(eval(valS)) * eval(unit1S)
313                    val2U = [q.cast_unit(eval(unit2S)) for q in val1U]
314                else:
315                    cmdS = varS + "= " + valS
316                    exec(cmdS, globals(), locals())
317                    valU = eval(varS)
318                    val1U = str(valU.cast_unit(eval(unit1S)))
319                    val2U = str(valU.cast_unit(eval(unit2S)))
320            else:
321                cmdS = varS + "= " + valS
322                exec(cmdS, globals(), locals())
323                valU = eval(varS)
324                val1U = str(valU)
325                val2U = str(valU)
326            valL.append([varS, val1U, val2U, descripS])
327
328        sys.stdout.flush()
329        old_stdout = sys.stdout
330        output = StringIO()
331        output.write(
332            tabulate(
333                valL, tablefmt=tblfmt, headers=hdreL,
334                showindex=False,  colalign=alignaL))
335        mdS = output.getvalue()
336        sys.stdout = old_stdout
337        sys.stdout.flush()
338
339        self.localD.update(locals())
340        print("\n" + mdS+"\n")
341        return mdS
