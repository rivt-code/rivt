 38# temperature - relative degree size, not offset ------------------------
 39K = new_unit("K", 0, "kelvin")
 40CELSIUS = new_unit("deg C", K, "degree Celsius")
 41FAHR = new_unit("degF", K * 9.0 / 5, "degree Fahrenheit")
 42# time
 43SEC = S = new_unit("s", 0, "second")
 44HZ = new_unit("Hz", 1 / S, "hertz")
 45# length
 46M = new_unit("m", 0, "meter")
 47NM = new_unit("nm", 10**-9 * M, "nanometer")
 48UM = new_unit("um", 10**-6 * M, "micrometer")
 49MM = new_unit("mm", 10**-3 * M, "millimeter")
 50CM = new_unit("cm", 10**-2 * M, "centimeter")
 51DM = new_unit("dm", 10**-1 * M, "decimeter")
 52# mass
 53KG = new_unit("kg", 0, "kilogram")
 54GRAM = new_unit("gram", 10**-3 * KG, "gram")
 55RAD = new_unit("rad", M / M, "radian")
 56SR = new_unit("sr", M**2 / M**2, "steradian")
 57MOL = new_unit("mol", 0, "mole")
 58# force
 59N = new_unit("N", M * KG / S**2, "newton")
 60CD = new_unit("cd", 0, "candela")
 61LM = new_unit("lm", CD * SR, "lumen")
 62LX = new_unit("lx", LM / M**2, "lux")
 63# charge
 64J = new_unit("J", N * M, "joule")
 65W = new_unit("W", J / S, "watt")
 66A = new_unit("A", 0, "ampere")
 67MA = new_unit("mA", 10**-3 * A, "milliampere")
 68C = new_unit("C", S * A, "coulomb")
 69VO = new_unit("V", W / A, "volt")
 70F = new_unit("F", C / VO, "farad")
 71OHM = new_unit("ohm", VO / A, "ohm")
 72SIEMENS = new_unit("siemens", A / VO, "siemens")
 73WB = new_unit("Wb", VO * SIEMENS, "weber")
 74TS = new_unit("TS", WB / M**2, "tesla")
 75HENRY = new_unit("H", WB / A, "henry")
